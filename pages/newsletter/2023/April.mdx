function calculateAverage(arr) {
  let sum = 0;
  let count = 0;

  for (let i = 0; i < arr.length; i++) {
    if (arr[i] && typeof arr[i] === "number") {
      sum += arr[i];
      count++;
    }
  }

  return sum / count;
}
